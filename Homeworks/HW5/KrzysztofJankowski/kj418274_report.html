Task 1

Problem statement

We have a model $$ f(x_1, x_2) = (x_1 + x_2) ^ 2 $$ and $$ x_1, x_2 \sim U[-1, 1], \ x_1 = x_2 $$ We will calculate the PD profile for variable $x_1$ in this model.

Calculation

$$ g_1^{PD}(z) = E_{X_2}[z^2 + 2zx_2 + x_2^2] = $$ $$ = z^2 + 2zE_{X_2}[x_2] + E_{X_2}[x_2^2] = $$ $$ = z^2 + 2z \cdot 0 + \frac{1}{3} = z^2 + \frac{1}{3}$$ In last steps we used the fact that because $x_1 \sim U[-1, 1]$ then $E_{X_2}[x_2] = 0$ and $E_{X_2}[x_2^2] = \int_{-1}^{1} x_2^2 \cdot f_{X_2}(x_2) \,dx_2 = \int_{-1}^{1} \frac{x_2^2}{2} \,dx = \frac{1^3}{6} - \frac{(-1)^3}{6} = 1/3$

from matplotlib import pyplot as plt
from typing import Tuple


class MyProfile:

    def __init__(self, model: ScikitModel, X: pd.DataFrame, num_points: int = 50, figsize: Tuple[int, int] = (10, 3)):
        self.model = model
        self.num_points = num_points
        self.X = X
        self.figsize = figsize

    def cp_profile(self, observation: pd.DataFrame, variable: str) -> Tuple[np.ndarray, pd.DataFrame]:
        value = observation[variable].values[0]

        # value on x axis with also the value from the observation.
        test_values = np.linspace(self.X[variable].min(), self.X[variable].max(), self.num_points - 1)
        index_to_insert = np.searchsorted(test_values, value)
        test_values = np.insert(test_values, index_to_insert, value)

        new_observations = pd.concat([observation] * self.num_points, ignore_index=True)
        new_observations[variable] = test_values
        new_models_predictions = self.model.predict_proba(new_observations)[:, 1]
        return test_values, new_models_predictions

    def pdp_profile(self, variable: str, N: int = 30) -> tuple[np.ndarray, np.ndarray, np.ndarray]:
        test_values = np.linspace(self.X[variable].min(), self.X[variable].max(), self.num_points)
        # sample N observations from the dataset
        sampled_observations = self.X.sample(N)
        # iterate rows of sampled_observations
        cp_profiles = []
        for index, row in sampled_observations.iterrows():
            _, cp_profile = self.cp_profile(row.to_frame().T, variable)
            cp_profiles.append(cp_profile)

        # concatenate cp_profiles into one pandas
        cp_profiles = np.array(cp_profiles)
        pdp_profile = cp_profiles.mean(axis=0)

        return test_values, cp_profiles, pdp_profile

    def plot_cp_profile(self, observation: pd.DataFrame, variable: str,
                        values: Tuple[np.ndarray, pd.DataFrame]) -> None:
        # set figsize 
        plt.figure(figsize=self.figsize)
        model_prediction = self.model.predict_proba(observation)[:, 1]
        value = observation[variable].values[0]
        plt.plot(values[0], values[1], label='CP profile')
        plt.plot(value, model_prediction, 'ro', label='Observation')
        plt.title(f'CP profile for variable {variable}.')
        # add legend
        plt.legend()

    def plot_pdp_profile(self, variable: str, values: Tuple[np.ndarray, np.ndarray, np.ndarray]) -> None:
        plt.figure(figsize=self.figsize)
        plt.plot(values[0], values[1].T, 'grey', alpha=0.4, linewidth=1)
        plt.plot(values[0], values[2], 'lightblue', linewidth=5, label='PDP profile')
        plt.title(f'PDP profile for variable {variable}.')
        plt.legend()
